025/**
026 * A ANSI output stream extracts ANSI escape codes written to 
027 * an output stream. 
028 *
029 * For more information about ANSI escape codes, see:
030 * http://en.wikipedia.org/wiki/ANSI_escape_code
031 *
032 * This class just filters out the escape codes so that they are not
033 * sent out to the underlying OutputStream.  Subclasses should
034 * actually perform the ANSI escape behaviors.
035 *
036 * @author <a href="http://hiramchirino.com">Hiram Chirino</a>
037 * @author Joris Kuipers
038 * @since 1.0
039 */
040public class AnsiOutputStream extends FilterOutputStream {
041
042    public static final byte[] REST_CODE = resetCode();
043
044    public AnsiOutputStream(OutputStream os) {
045        super(os);
046    }

048    private final static int MAX_ESCAPE_SEQUENCE_LENGTH = 100;
049    private final byte[] buffer = new byte[MAX_ESCAPE_SEQUENCE_LENGTH];
050    private int pos = 0;
051    private int startOfValue;
052    private final ArrayList<Object> options = new ArrayList<Object>();
053
054    private static final int LOOKING_FOR_FIRST_ESC_CHAR = 0;
055    private static final int LOOKING_FOR_SECOND_ESC_CHAR = 1;
056    private static final int LOOKING_FOR_NEXT_ARG = 2;
057    private static final int LOOKING_FOR_STR_ARG_END = 3;
058    private static final int LOOKING_FOR_INT_ARG_END = 4;
059    private static final int LOOKING_FOR_OSC_COMMAND = 5;
060    private static final int LOOKING_FOR_OSC_COMMAND_END = 6;
061    private static final int LOOKING_FOR_OSC_PARAM = 7;
062    private static final int LOOKING_FOR_ST = 8;
063
064    int state = LOOKING_FOR_FIRST_ESC_CHAR;
065
066    private static final int FIRST_ESC_CHAR = 27;
067    private static final int SECOND_ESC_CHAR = '[';
068    private static final int SECOND_OSC_CHAR = ']';
069    private static final int BEL = 7;
070    private static final int SECOND_ST_CHAR = '\\';
071
072    // TODO: implement to get perf boost: public void write(byte[] b, int off, int len)
073
074    public void write(int data) throws IOException {
075        switch (state) {
076            case LOOKING_FOR_FIRST_ESC_CHAR:
077                if (data == FIRST_ESC_CHAR) {
078                    buffer[pos++] = (byte) data;
079                    state = LOOKING_FOR_SECOND_ESC_CHAR;
080                } else {
081                    out.write(data);
082                }
083                break;
084
085            case LOOKING_FOR_SECOND_ESC_CHAR:
086                buffer[pos++] = (byte) data;
087                if (data == SECOND_ESC_CHAR) {
088                    state = LOOKING_FOR_NEXT_ARG;
089                } else if (data == SECOND_OSC_CHAR) {
090                    state = LOOKING_FOR_OSC_COMMAND;
091                } else {
092                    reset(false);
093                }
094                break;
095
096            case LOOKING_FOR_NEXT_ARG:
097                buffer[pos++] = (byte) data;
098                if ('"' == data) {
099                    startOfValue = pos - 1;
100                    state = LOOKING_FOR_STR_ARG_END;
101                } else if ('0' <= data && data <= '9') {
102                    startOfValue = pos - 1;
103                    state = LOOKING_FOR_INT_ARG_END;
104                } else if (';' == data) {
105                    options.add(null);
106                } else if ('?' == data) {
107                    options.add('?');
108                } else if ('=' == data) {
109                    options.add('=');
110                } else {
111                    reset(processEscapeCommand(options, data));
112                }
113                break;
114            default:
115                break;
116
117            case LOOKING_FOR_INT_ARG_END:
118                buffer[pos++] = (byte) data;
119                if (!('0' <= data && data <= '9')) {
120                    String strValue = new String(buffer, startOfValue, (pos - 1) - startOfValue, "UTF-8");
121                    Integer value = new Integer(strValue);
122                    options.add(value);
123                    if (data == ';') {
124                        state = LOOKING_FOR_NEXT_ARG;
125                    } else {
126                        reset(processEscapeCommand(options, data));
127                    }
128                }
129                break;
130
131            case LOOKING_FOR_STR_ARG_END:
132                buffer[pos++] = (byte) data;
133                if ('"' != data) {
134                    String value = new String(buffer, startOfValue, (pos - 1) - startOfValue, "UTF-8");
135                    options.add(value);
136                    if (data == ';') {
137                        state = LOOKING_FOR_NEXT_ARG;
138                    } else {
139                        reset(processEscapeCommand(options, data));
140                    }
141                }
142                break;
143
144            case LOOKING_FOR_OSC_COMMAND:
145                buffer[pos++] = (byte) data;
146                if ('0' <= data && data <= '9') {
147                    startOfValue = pos - 1;
148                    state = LOOKING_FOR_OSC_COMMAND_END;
149                } else {
150                    reset(false);
151                }
152                break;
153
154            case LOOKING_FOR_OSC_COMMAND_END:
155                buffer[pos++] = (byte) data;
156                if (';' == data) {
157                    String strValue = new String(buffer, startOfValue, (pos - 1) - startOfValue, "UTF-8");
158                    Integer value = new Integer(strValue);
159                    options.add(value);
160                    startOfValue = pos;
161                    state = LOOKING_FOR_OSC_PARAM;
162                } else if ('0' <= data && data <= '9') {
163                    // already pushed digit to buffer, just keep looking
164                } else {
165                    // oops, did not expect this
166                    reset(false);
167                }
168                break;
169
170            case LOOKING_FOR_OSC_PARAM:
171                buffer[pos++] = (byte) data;
172                if (BEL == data) {
173                    String value = new String(buffer, startOfValue, (pos - 1) - startOfValue, "UTF-8");
174                    options.add(value);
175                    reset(processOperatingSystemCommand(options));
176                } else if (FIRST_ESC_CHAR == data) {
177                    state = LOOKING_FOR_ST;
178                } else {
179                    // just keep looking while adding text
180                }
181                break;
182
183            case LOOKING_FOR_ST:
184                buffer[pos++] = (byte) data;
185                if (SECOND_ST_CHAR == data) {
186                    String value = new String(buffer, startOfValue, (pos - 2) - startOfValue, "UTF-8");
187                    options.add(value);
188                    reset(processOperatingSystemCommand(options));
189                } else {
190                    state = LOOKING_FOR_OSC_PARAM;
191                }
192                break;
193        }
194
195        // Is it just too long?
196        if (pos >= buffer.length) {
197            reset(false);
198        }
199    }

201    /**
202     * Resets all state to continue with regular parsing
203     * @param skipBuffer if current buffer should be skipped or written to out
204     * @throws IOException
205     */
206    private void reset(boolean skipBuffer) throws IOException {
207        if (!skipBuffer) {
208            out.write(buffer, 0, pos);
209        }
210        pos = 0;
211        startOfValue = 0;
212        options.clear();
213        state = LOOKING_FOR_FIRST_ESC_CHAR;
214    }

216    /**
217     * Helper for processEscapeCommand() to iterate over integer options
218     * @param  optionsIterator  the underlying iterator
219     * @throws IOException      if no more non-null values left
220     */
221    private int getNextOptionInt(Iterator<Object> optionsIterator) throws IOException {
222        for (;;) {
223            if (!optionsIterator.hasNext())
224                throw new IllegalArgumentException();
225            Object arg = optionsIterator.next();
226            if (arg != null)
227                return (Integer) arg;
228        }
229    }

231    /**
232     *
233     * @param options
234     * @param command
235     * @return true if the escape command was processed.
236     */
237    private boolean processEscapeCommand(ArrayList<Object> options, int command) throws IOException {
238        try {
239            switch (command) {
240                case 'A':
241                    processCursorUp(optionInt(options, 0, 1));
242                    return true;
243                case 'B':
244                    processCursorDown(optionInt(options, 0, 1));
245                    return true;
246                case 'C':
247                    processCursorRight(optionInt(options, 0, 1));
248                    return true;
249                case 'D':
250                    processCursorLeft(optionInt(options, 0, 1));
251                    return true;
252                case 'E':
253                    processCursorDownLine(optionInt(options, 0, 1));
254                    return true;
255                case 'F':
256                    processCursorUpLine(optionInt(options, 0, 1));
257                    return true;
258                case 'G':
259                    processCursorToColumn(optionInt(options, 0));
260                    return true;
261                case 'H':
262                case 'f':
263                    processCursorTo(optionInt(options, 0, 1), optionInt(options, 1, 1));
264                    return true;
265                case 'J':
266                    processEraseScreen(optionInt(options, 0, 0));
267                    return true;
268                case 'K':
269                    processEraseLine(optionInt(options, 0, 0));
270                    return true;
271                case 'S':
272                    processScrollUp(optionInt(options, 0, 1));
273                    return true;
274                case 'T':
275                    processScrollDown(optionInt(options, 0, 1));
276                    return true;
277                case 'm':
278                    // Validate all options are ints...
279                    for (Object next : options) {
280                        if (next != null && next.getClass() != Integer.class) {
281                            throw new IllegalArgumentException();
282                        }
283                    }
284
285                    int count = 0;
286                    Iterator<Object> optionsIterator = options.iterator();
287                    while (optionsIterator.hasNext()) {
288                        Object next = optionsIterator.next();
289                        if (next != null) {
290                            count++;
291                            int value = (Integer) next;
292                            if (30 <= value && value <= 37) {
293                                processSetForegroundColor(value - 30);
294                            } else if (40 <= value && value <= 47) {
295                                processSetBackgroundColor(value - 40);
296                            } else if (90 <= value && value <= 97) {
297                                processSetForegroundColor(value - 90, true);
298                            } else if (100 <= value && value <= 107) {
299                                processSetBackgroundColor(value - 100, true);
300                            } else if (value == 38 || value == 48) {
301                                // extended color like `esc[38;5;<index>m` or `esc[38;2;<r>;<g>;<b>m`
302                                int arg2or5 = getNextOptionInt(optionsIterator);
303                                if (arg2or5 == 2) {
304                                    // 24 bit color style like `esc[38;2;<r>;<g>;<b>m`
305                                    int r = getNextOptionInt(optionsIterator);
306                                    int g = getNextOptionInt(optionsIterator);
307                                    int b = getNextOptionInt(optionsIterator);
308                                    if (r >= 0 && r <= 255 && g >= 0 && g <= 255 && b >= 0 && b <= 255) {
309                                        if (value == 38)
310                                            processSetForegroundColorExt(r, g, b);
311                                        else
312                                            processSetBackgroundColorExt(r, g, b);
313                                    } else {
314                                        throw new IllegalArgumentException();
315                                    }
316                                }
317                                else if (arg2or5 == 5) {
318                                    // 256 color style like `esc[38;5;<index>m`
319                                    int paletteIndex = getNextOptionInt(optionsIterator);
320                                    if (paletteIndex >= 0 && paletteIndex <= 255) {
321                                        if (value == 38)
322                                            processSetForegroundColorExt(paletteIndex);
323                                        else
324                                            processSetBackgroundColorExt(paletteIndex);
325                                    } else {
326                                        throw new IllegalArgumentException();
327                                    }
328                                }
329                                else {
330                                    throw new IllegalArgumentException();
331                                }
332                            } else {
333                                switch (value) {
334                                    case 39:
335                                        processDefaultTextColor();
336                                        break;
337                                    case 49:
338                                        processDefaultBackgroundColor();
339                                        break;
340                                    case 0:
341                                        processAttributeRest();
342                                        break;
343                                    default:
344                                        processSetAttribute(value);
345                                }
346                            }
347                        }
348                    }
349                    if (count == 0) {
350                        processAttributeRest();
351                    }
352                    return true;
353                case 's':
354                    processSaveCursorPosition();
355                    return true;
356                case 'u':
357                    processRestoreCursorPosition();
358                    return true;
359
360                default:
361                    if ('a' <= command && 'z' <= command) {
362                        processUnknownExtension(options, command);
363                        return true;
364                    }
365                    if ('A' <= command && 'Z' <= command) {
366                        processUnknownExtension(options, command);
367                        return true;
368                    }
369                    return false;
370            }
371        } catch (IllegalArgumentException ignore) {
372        }
373        return false;
374    }

376    /**
377     *
378     * @param options
379     * @return true if the operating system command was processed.
380     */
381    private boolean processOperatingSystemCommand(ArrayList<Object> options) throws IOException {
382        int command = optionInt(options, 0);
383        String label = (String) options.get(1);
384        // for command > 2 label could be composed (i.e. contain ';'), but we'll leave
385        // it to processUnknownOperatingSystemCommand implementations to handle that
386        try {
387            switch (command) {
388                case 0:
389                    processChangeIconNameAndWindowTitle(label);
390                    return true;
391                case 1:
392                    processChangeIconName(label);
393                    return true;
394                case 2:
395                    processChangeWindowTitle(label);
396                    return true;
397
398                default:
399                    // not exactly unknown, but not supported through dedicated process methods:
400                    processUnknownOperatingSystemCommand(command, label);
401                    return true;
402            }
403        } catch (IllegalArgumentException ignore) {
404        }
405        return false;
406    }

547    private int optionInt(ArrayList<Object> options, int index) {
548        if (options.size() <= index)
549            throw new IllegalArgumentException();
550        Object value = options.get(index);
551        if (value == null)
552            throw new IllegalArgumentException();
553        if (!value.getClass().equals(Integer.class))
554            throw new IllegalArgumentException();
555        return (Integer) value;
556    }
557
558    private int optionInt(ArrayList<Object> options, int index, int defaultValue) {
559        if (options.size() > index) {
560            Object value = options.get(index);
561            if (value == null) {
562                return defaultValue;
563            }
564            return (Integer) value;
565        }
566        return defaultValue;
567    }

569    @Override
570    public void close() throws IOException {
571        write(REST_CODE);
572        flush();
573        super.close();
574    }

576    static private byte[] resetCode() {
577        try {
578            return new Ansi().reset().toString().getBytes("UTF-8");
579        } catch (UnsupportedEncodingException e) {
580            throw new RuntimeException(e);
581        }
582    }
