016package org.fusesource.jansi.io;
017
018import java.io.IOException;
019import java.io.OutputStream;
020import java.util.ArrayList;
021import java.util.Iterator;
022
023import org.fusesource.jansi.AnsiColors;
024
025/**
026 * Ansi processor to process color conversion if needed.
027 */
028public class ColorsAnsiProcessor extends AnsiProcessor {
029
030    private final AnsiColors colors;
031
032    public ColorsAnsiProcessor(OutputStream os, AnsiColors colors) {
033        super(os);
034        this.colors = colors;
035    }
036
037    @Override
038    protected boolean processEscapeCommand(ArrayList<Object> options, int command) throws IOException {
039        if (command == 'm' && (colors == AnsiColors.Colors256 || colors == AnsiColors.Colors16)) {
040            // Validate all options are ints...
041            boolean has38or48 = false;
042            for (Object next : options) {
043                if (next != null && next.getClass() != Integer.class) {
044                    throw new IllegalArgumentException();
045                }
046                Integer value = (Integer) next;
047                has38or48 |= value == 38 || value == 48;
048            }
049            // SGR commands do not contain an extended color, so just transfer the buffer
050            if (!has38or48) {
051                return false;
052            }
053            StringBuilder sb = new StringBuilder(32);
054            sb.append('\033').append('[');
055            boolean first = true;
056            Iterator<Object> optionsIterator = options.iterator();
057            while (optionsIterator.hasNext()) {
058                Object next = optionsIterator.next();
059                if (next != null) {
060                    int value = (Integer) next;
061                    if (value == 38 || value == 48) {
062                        // extended color like `esc[38;5;<index>m` or `esc[38;2;<r>;<g>;<b>m`
063                        int arg2or5 = getNextOptionInt(optionsIterator);
064                        if (arg2or5 == 2) {
065                            // 24 bit color style like `esc[38;2;<r>;<g>;<b>m`
066                            int r = getNextOptionInt(optionsIterator);
067                            int g = getNextOptionInt(optionsIterator);
068                            int b = getNextOptionInt(optionsIterator);
069                            if (colors == AnsiColors.Colors256) {
070                                int col = Colors.roundRgbColor(r, g, b, 256);
071                                if (!first) {
072                                    sb.append(';');
073                                }
074                                first = false;
075                                sb.append(value);
076                                sb.append(';');
077                                sb.append(5);
078                                sb.append(';');
079                                sb.append(col);
080                            } else {
081                                int col = Colors.roundRgbColor(r, g, b, 16);
082                                if (!first) {
083                                    sb.append(';');
084                                }
085                                first = false;
086                                sb.append(value == 38 ? col >= 8 ? 90 + col - 8 : 30 + col : col >= 8 ? 100 + col - 8 : 40 + col);
087                            }
088                        }
089                        else if (arg2or5 == 5) {
090                            // 256 color style like `esc[38;5;<index>m`
091                            int paletteIndex = getNextOptionInt(optionsIterator);
092                            if (colors == AnsiColors.Colors256) {
093                                if (!first) {
094                                    sb.append(';');
095                                }
096                                first = false;
097                                sb.append(value);
098                                sb.append(';');
099                                sb.append(5);
100                                sb.append(';');
101                                sb.append(paletteIndex);
102                            } else {
103                                int col = Colors.roundColor(paletteIndex, 16);
104                                if (!first) {
105                                    sb.append(';');
106                                }
107                                first = false;
108                                sb.append(value == 38 ? col >= 8 ? 90 + col - 8 : 30 + col : col >= 8 ? 100 + col - 8 : 40 + col);
109                            }
110                        }
111                        else {
112                            throw new IllegalArgumentException();
113                        }
114                    } else {
115                        if (!first) {
116                            sb.append(';');
117                        }
118                        first = false;
119                        sb.append(value);
120                    }
121                }
122            }
123            sb.append('m');
124            os.write(sb.toString().getBytes());
125            return true;
126
127        } else {
128            return false;
129        }
130    }
131
132    @Override
133    protected boolean processOperatingSystemCommand(ArrayList<Object> options) {
134        return false;
135    }
136
137    @Override
138    protected boolean processCharsetSelect(ArrayList<Object> options) {
139        return false;
140    }
141
142}
